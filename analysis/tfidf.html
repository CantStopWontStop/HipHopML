song_words <- lyr_df |> 
    drop_na(hip_hop_rnb) |> 
    unnest_tokens(word, lyrics)|>
    add_count(song, artist, name = "total_words") |> 
    anti_join(stop_words) |> 
    group_by(song, artist, hip_hop_rnb, genre, total_words) |> 
    count(word, sort = TRUE) |> 
    ungroup()
