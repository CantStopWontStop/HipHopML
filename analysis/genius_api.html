Load Songs

all_songs <- read_csv('../data/all_songs.csv')|> 
    mutate(across(everything(),
                  str_remove_all,
                  '\"')) |> 
    separate(artist, 
             into = c("artist","features"),
             fill = "right",
             sep="featuring ") |> 
    select(song, artist, hip_hop_rnb, genre) |> 
    drop_na() |> 
    group_by(song, artist) |> 
    slice(1) |> 
    ungroup()

Get Lyrics

Query Genius API

lyr_df <- all_songs |>
    rowwise() |>
    mutate(
        tryCatch(
            get_lyrics_search(
                artist_name = artist,
                song_title  = song
                ) |>
                summarise(
                    lyrics  = str_c(line,
                                    collapse = " ")
                    ),
            error = function(w)
                return(NA)
            ) 
       ) |> 
    unnest(lyrics) |> 
    select(song, artist, hip_hop_rnb,lyrics, genre)
