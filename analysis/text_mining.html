plot_by_year <- function(year){
    words_by_year_df |> 
        group_by(rel_year) |> 
        slice_max(n, n = 100) |> 
        filter(rel_year == year) |> 
        ggplot(aes(label = word, size = n)) +
            geom_text_wordcloud(color = "#145389") +
            scale_size_area(max_size = 14) +
            facet_wrap(~rel_year,scales = "free" ) +
            theme_minimal()
    
}
